function init() {
	$('.smallthumbs li').each(function() {
		id = $('a', this).attr('href').substring(1);
		if (isNaN(parseInt(id))) {
			return;
		}
		url = 'https://s3.zerochan.net/75/';
		if (id % 50 < 10) {
			url += '0';
		}
		url += id % 50 + '/';
		if ((id / 50) % 50 < 10) {
			url += '0';
		}
		url += Math.floor(id / 50) % 50;
		url += '/' + id
		url += '.jpg'
		$('a', this).css('background-image', 'url(' + url + ')');
	});
}
